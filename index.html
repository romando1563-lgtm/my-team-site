function isOverdue(g){ return g.deadline && !g.done && new Date(g.deadline) < new Date().setHours(0,0,0,0); }

/* --- Drag & Drop --- */

let dragged = null;

function attachDrag() {

  document.querySelectorAll('.emp').forEach(el => {

    el.addEventListener('dragstart', () => { dragged = { emp: el.dataset.emp, group: el.dataset.group }; el.classList.add('dragging'); });

    el.addEventListener('dragend', () => el.classList.remove('dragging'));

  });

}

function onDragOver(e){ e.preventDefault(); e.currentTarget.classList.add('dragover'); }

function onDragLeave(e){ e.currentTarget.classList.remove('dragover'); }

function onDrop(e, targetGroup){

  e.preventDefault(); e.currentTarget.classList.remove('dragover');

  if(!dragged || dragged.group === targetGroup) return;

  const fromG = data.groups.find(g=>g.id===dragged.group);

  const toG = data.groups.find(g=>g.id===targetGroup);

  const idx = fromG.employees.findIndex(x=>x.id===dragged.emp);

  const [emp] = fromG.employees.splice(idx,1);

  toG.employees.push(emp);

  dragged = null;

  render();

  saveDataToDB();

}

function openBossModal(){ document.getElementById('bossName').value = data.boss?.name||''; document.getElementById('bossRole').value = data.boss?.role||''; document.getElementById('bossOverlay').classList.add('show'); }

function closeBossModal(){ document.getElementById('bossOverlay').classList.remove('show'); }

function saveBoss(){ const name = document.getElementById('bossName').value.trim(); if(!name) return alert('Имя обязательно'); data.boss = { name, role: document.getElementById('bossRole').value.trim() }; closeBossModal(); render(); saveDataToDB(); }


function openEmpModal(groupId){ editing = { groupId, empId:null }; document.getElementById('empModalTitle').textContent = 'Новый'; ['empName','empRole','empSchedule','empDuties'].forEach(id=>document.getElementById(id).value=''); document.getElementById('empStatus').value='member'; document.getElementById('empOverlay').classList.add('show'); }

function editEmp(groupId, empId){ editing = { groupId, empId }; const e = data.groups.find(g=>g.id===groupId).employees.find(x=>x.id===empId); document.getElementById('empModalTitle').textContent = 'Ред.'; document.getElementById('empName').value=e.name; document.getElementById('empRole').value=e.role; document.getElementById('empSchedule').value=e.schedule; document.getElementById('empStatus').value=e.status; document.getElementById('empDuties').value=e.duties.join('\n'); document.getElementById('empOverlay').classList.add('show'); }

function saveEmp(){ const name = document.getElementById('empName').value.trim(); if(!name) return alert('Имя обязательно'); const g = data.groups.find(g=>g.id===editing.groupId); const obj = { name, role: document.getElementById('empRole').value.trim(), schedule: document.getElementById('empSchedule').value.trim(), status: document.getElementById('empStatus').value, duties: document.getElementById('empDuties').value.split('\n').map(s=>s.trim()).filter(Boolean) }; if(editing.empId){ Object.assign(g.employees.find(x=>x.id===editing.empId), obj); } else { obj.id='e'+Date.now(); obj.goals=[]; g.employees.push(obj); } closeEmpModal(); render(); saveDataToDB(); }

function closeEmpModal(){ document.getElementById('empOverlay').classList.remove('show'); }

function deleteEmp(groupId, empId){ if(!confirm('Удалить?')) return; const g = data.groups.find(g=>g.id===groupId); g.employees = g.employees.filter(x=>x.id!==empId); render(); saveDataToDB(); }


function openDutyModal(groupId, empId){ dutyTarget = { groupId, empId }; document.getElementById('dutyText').value=''; document.getElementById('dutyOverlay').classList.add('show'); }

function closeDutyModal(){ document.getElementById('dutyOverlay').classList.remove('show'); }

function saveDuty(){ const text = document.getElementById('dutyText').value.trim(); if(!text) return; data.groups.find(g=>g.id===dutyTarget.groupId).employees.find(x=>x.id===dutyTarget.empId).duties.push(text); closeDutyModal(); render(); saveDataToDB(); }

function removeDuty(groupId, empId, i){ data.groups.find(g=>g.id===groupId).employees.find(x=>x.id===empId).duties.splice(i,1); render(); saveDataToDB(); }


function getGoalArray(type, groupId, empId){ const g = data.groups.find(g=>g.id===groupId); return type==='team' ? g.goals : g.employees.find(x=>x.id===empId).goals; }

function openGoalModal(type, groupId, empId=null){ goalTarget = { type, groupId, empId, goalId:null }; const g = data.groups.find(g=>g.id===groupId); const who = type==='team' ? g.name : g.employees.find(x=>x.id===empId).name; document.getElementById('goalModalTitle').textContent = 'Цель: ' + who; document.getElementById('goalText').value=''; document.getElementById('goalPrio').value='mid'; document.getElementById('goalDeadline').value=''; document.getElementById('goalProgress').value=0; document.getElementById('goalOverlay').classList.add('show'); }

function editGoal(type, groupId, empId, goalId){ goalTarget = { type, groupId, empId, goalId }; const gl = getGoalArray(type, groupId, empId).find(x=>x.id===goalId); document.getElementById('goalModalTitle').textContent = 'Ред. цель'; document.getElementById('goalText').value=gl.text; document.getElementById('goalPrio').value=gl.prio; document.getElementById('goalDeadline').value=gl.deadline||''; document.getElementById('goalProgress').value=gl.progress; document.getElementById('goalOverlay').classList.add('show'); }

function saveGoal(){ const text = document.getElementById('goalText').value.trim(); if(!text) return alert('Текст цели'); const arr = getGoalArray(goalTarget.type, goalTarget.groupId, goalTarget.empId); const progress = parseInt(document.getElementById('goalProgress').value)||0; const obj = { text, prio: document.getElementById('goalPrio').value, deadline: document.getElementById('goalDeadline').value, progress, done: progress>=100 }; if(goalTarget.goalId){ Object.assign(arr.find(x=>x.id===goalTarget.goalId), obj); } else { obj.id='gl'+Date.now(); arr.push(obj); } closeGoalModal(); render(); saveDataToDB(); }

function closeGoalModal(){ document.getElementById('goalOverlay').classList.remove('show'); }

function deleteGoal(type, groupId, empId, goalId){ if(!confirm('Удалить?')) return; const arr = getGoalArray(type, groupId, empId); const i = arr.findIndex(x=>x.id===goalId); arr.splice(i,1); render(); saveDataToDB(); }

function toggleGoal(type, groupId, empId, goalId){ const gl = getGoalArray(type, groupId, empId).find(x=>x.id===goalId); gl.done = !gl.done; if(gl.done) gl.progress = 100; render(); saveDataToDB(); }

function setGoalProgress(type, groupId, empId, goalId, val){ const gl = getGoalArray(type, groupId, empId).find(x=>x.id===goalId); gl.progress = parseInt(val); gl.done = gl.progress>=100; render(); }


function openGroupModal(){ editingGroupId = null; document.getElementById('groupModalTitle').textContent = 'Новая группа'; document.getElementById('groupSaveBtn').textContent = 'Создать'; document.getElementById('groupName').value=''; document.getElementById('groupColor').value='#4f46e5'; document.getElementById('groupOverlay').classList.add('show'); }

function editGroup(id){ const g = data.groups.find(x=>x.id===id); if(!g) return; editingGroupId = id; document.getElementById('groupModalTitle').textContent = 'Ред. группу'; document.getElementById('groupSaveBtn').textContent = 'Сохранить'; document.getElementById('groupName').value = g.name; document.getElementById('groupColor').value = g.color; document.getElementById('groupOverlay').classList.add('show'); }

function closeGroupModal(){ document.getElementById('groupOverlay').classList.remove('show'); }

function saveGroup(){ const name = document.getElementById('groupName').value.trim(); if(!name) return alert('Название'); const color = document.getElementById('groupColor').value; if(editingGroupId){ const g = data.groups.find(x=>x.id===editingGroupId); g.name = name; g.color = color; } else { data.groups.push({ id:'g'+Date.now(), name, color, goals:[], employees:[] }); } closeGroupModal(); render(); saveDataToDB(); }

function deleteGroup(id){ if(!confirm('Удалить группу?')) return; data.groups = data.groups.filter(g=>g.id!==id); render(); saveDataToDB(); }


function exportData(){ const blob = new Blob([JSON.stringify(data,null,2)],{type:'application/json'}); const a = document.createElement('a'); a.href = URL.createObjectURL(blob); a.download = 'data.json'; a.click(); }

function triggerImport(){ document.getElementById('importInput').click(); }

function importData(event){ const file = event.target.files[0]; if(!file) return; const reader = new FileReader(); reader.onload = (e) => { try { const parsed = JSON.parse(e.target.result); if(!parsed.groups) return alert('Ошибка формата'); data = parsed; render(); saveDataToDB(); alert('Загружено!'); } catch(err){ alert('Ошибка'); } }; reader.readAsText(file); event.target.value = ''; }

function exportHTML(){ const doc = document.documentElement.cloneNode(true); doc.querySelectorAll('.overlay.show').forEach(o => o.classList.remove('show')); const scripts = doc.querySelectorAll('script'); scripts.forEach(sc => { if (sc.textContent.includes('let data = {')) { const dataStr = JSON.stringify(data, null, 2); sc.textContent = sc.textContent.replace(/let data = \{[\s\S]*?\n\};/, 'let data = ' + dataStr + ';'); } }); const html = '<!DOCTYPE html>\n' + doc.outerHTML; const blob = new Blob([html], { type:'text/html' }); const a = document.createElement('a'); a.href = URL.createObjectURL(blob); a.download = `dashboard-${new Date().toISOString().slice(0,10)}.html`; a.click(); }
